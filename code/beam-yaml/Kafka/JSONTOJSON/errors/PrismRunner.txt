        topic: "m-topic"

        consumer_config:
          group.id: beam-json-002

    # 2) Uppercase the "name" field and append as "name_upper"

        language: python
        append: true
        fields:

        topic: "n-topic"
        bootstrap_servers: "localhost:9092"
        format: JSON
